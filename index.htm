<script>

const STORE="petApp";

 

/* SAVE/LOAD */

function save(){

    const data={meds:[],vets:[]};

 

    document.querySelectorAll("#medTable tbody tr").forEach(r=>{

        data.meds.push([...r.querySelectorAll("input")].map(i=>i.type==="checkbox"?i.checked:i.value));

    });

 

    document.querySelectorAll("#vetTable tbody tr").forEach(r=>{

        data.vets.push([...r.querySelectorAll("input")].map(i=>i.value));

    });

 

    localStorage.setItem(STORE,JSON.stringify(data));

}

 

function load(){

    const d=JSON.parse(localStorage.getItem(STORE)||"{}");

    (d.meds||[]).forEach(addMedRow);

    (d.vets||[]).forEach(addVetRow);

}

 

/* MED */

function addMedRow(data=[]){

    const r=document.createElement("tr");

    const fields=["text","text","text","number","datetime-local","text","checkbox"];

 

    fields.forEach((t,i)=>{

        let td=document.createElement("td");

        let input=document.createElement("input");

        input.type=t;

 

        if(t==="checkbox") input.checked=data[i]||false;

        else input.value=data[i]||"";

 

        input.oninput=()=>{update(r);save();};

 

        td.appendChild(input);

        r.appendChild(td);

    });

 

    let next=document.createElement("td");

    r.insertBefore(next,r.children[5]);

 

    let del=document.createElement("td");

    let b=document.createElement("button");

    b.textContent="X";

    b.onclick=()=>{r.remove();save();};

    del.appendChild(b);

    r.appendChild(del);

 

    document.querySelector("#medTable tbody").appendChild(r);

    update(r);

}

 

function update(r){

    const i=r.querySelectorAll("input");

    const freq=parseFloat(i[3].value);

    const last=i[4].value;

    const nextCell=r.children[5];

 

    if(freq && last){

        const next=new Date(new Date(last).getTime()+freq*3600000);

        nextCell.textContent=next.toLocaleString();

 

        r.classList.remove("overdue","given");

        if(i[6].checked) r.classList.add("given");

        else if(next<new Date()) r.classList.add("overdue");

    }

}

 

/* VET */

function addVetRow(data=[]){

    const r=document.createElement("tr");

    ["text","date","text"].forEach((t,i)=>{

        let td=document.createElement("td");

        let input=document.createElement("input");

        input.type=t;

        input.value=data[i]||"";

        input.oninput=save;

        td.appendChild(input);

        r.appendChild(td);

    });

 

    let del=document.createElement("td");

    let b=document.createElement("button");

    b.textContent="X";

    b.onclick=()=>{r.remove();save();};

    del.appendChild(b);

    r.appendChild(del);

 

    document.querySelector("#vetTable tbody").appendChild(r);

}

 

/* ICS */

function exportICS(){

    let events=[];

    document.querySelectorAll("#medTable tbody tr").forEach(r=>{

        const i=r.querySelectorAll("input");

        if(r.children[5].innerText){

            const d=new Date(r.children[5].innerText).toISOString().replace(/[-:]/g,"").split(".")[0];

            events.push(`BEGIN:VEVENT\nSUMMARY:${i[0].value} ${i[1].value}\nDTSTART:${d}\nEND:VEVENT`);

        }

    });

 

    const ics=`BEGIN:VCALENDAR\n${events.join("\n")}\nEND:VCALENDAR`;

    const a=document.createElement("a");

    a.href=URL.createObjectURL(new Blob([ics]));

    a.download="calendar.ics";

    a.click();

}

 

/* NOTIFICATIONS */

function enableNotifications(){Notification.requestPermission();}

 

/* SERVICE WORKER */

if("serviceWorker" in navigator){

    navigator.serviceWorker.register("sw.js");

}

 

load();

setInterval(()=>document.querySelectorAll("#medTable tbody tr").forEach(update),60000);
